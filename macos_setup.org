#+title: Macos Setup

* Change keybehavior
Change CapsLock to Control (keyboard ? keyboard shortcuts > Modifier Keys)
* Install Homebrew
** Install
#+begin_src shell
/bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/master/install.sh)"
#+end_src
** Add to PATH
#+begin_src shell
echo 'eval "(/opt/homebrew/bin/brew shellenv)"' >> /Users/$(whoami)/.zprofile
#+end_src
* Setup ssh
** Generate key
#+begin_src shell :noweb yes
ssh-keygen -t ed25519 -C "kay.freyer@icloud.com"
# - ssh-keygen -t ed25519 -C "kay.freyer@icloud.com"
#   - save under /Users/kaypro/.ssh/<<system_info.org:HOSTNAME_mac()>>-<<system_info.org:OS()>>_rsa
#   - add passphrase to lastPass {username}{PC}{OS}
#+end_src
- ssh-keygen -t ed25519 -C "kay.freyer@icloud.com"
  - save under .ssh/{username}{PC}{OS}_rsa
  - add passphrase to lastPass {username}{PC}{OS}
** Adding ssh_key to the ssh-agent
#+begin_src shell :noweb yes
ssh-add --apple-use-keychain /Users/kaypro/.ssh/<<system_info.org:HOSTNAME_mac()>>-<<system_info.org:OS()>>_rsa
touch .ssh/config
echo "Host *
    UseKeychain yes
    AddKeysToAgent yes
    IdentityFile ~/.ssh/<<system_info.org:HOSTNAME_mac()>>-<<system_info.org:OS()>>_rsa" >> .ssh/config
#+end_src
** Add pubic key to github
- cat ~/.ssh/id_ed25519.pub
- add ssh key on github

* Install Fonts
#+begin_src shell
brew tap homebrew/cask-fonts
brew install font-jetbrains-mono-nerd-font
#+end_src
* Install Emacs
** Dependencies
#+begin_src shell
brew install git ripgrep
brew install coreutils fd
xcode-select --install
#+end_src
** Emacs-mac
#+begin_src shell
brew tap railwaycat/emacsmacport
brew install emacs-mac --with-modules
ln -s /usr/local/opt/emacs-mac/Emacs.app /Applications/Emacs.app
#+end_src
* Install Doom Emacs
#+begin_src shell
git clone --depth 1 https://github.com/doomemacs/doomemacs ~/.config/emacs
PATH=$PATH:~/.config/emacs/bin
doom install
#+end_src
