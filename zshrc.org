#+title: Zshrc
#+STARTUP: overview
#+PROPERTY: header-args:shell :tangle ~/.dotfiles/.zshrc :mkdirp yes
#+INCLUDE: "~/.dotfiles/system_info.org"

* Adding ssh_key to the ssh-agent and handled by keychain
#+begin_src shell :noweb yes
eval $(keychain --eval ~/.ssh/<<system_info.org:HOSTNAME()>>-<<system_info.org:OS()>>_rsa)
#+end_src
* Export Path to .oh-my-zsh
#+begin_src shell
# Path to your oh-my-zsh installation.
export ZSH="$HOME/.oh-my-zsh"
#+end_src

* Add /emacs/bin to PATH
#+begin_src shell
PATH=$PATH:~/.config/emacs/bin
#+end_src
* Set Theme
** Standard
#+begin_src shell :tangle no
# Set name of the theme to load --- if set to "random", it will
# load a random theme each time oh-my-zsh is loaded, in which case,
# to know which specific one was loaded, run: echo $RANDOM_THEME
# See https://github.com/ohmyzsh/ohmyzsh/wiki/Themes
ZSH_THEME="robbyrussell"
#+end_src

* Setup Python
** Pyenv
#+begin_src shell
export PYENV_ROOT="$HOME/.pyenv"
export PATH="$PYENV_ROOT/bin:$PATH"
if command -v pyenv 1>/dev/null 2>&1; then
 eval "$(pyenv init --path)"
fi
#+end_src
